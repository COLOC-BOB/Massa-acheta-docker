import requests
import json

api_url = "https://mainnet.massa.net/api/v2"
block_id = "B12Nq4qdcG1e9Psuwi5akiAV6qQyVKw3TNjyBQDUCiazkFfTRCrG"

payload = {
    "jsonrpc": "2.0",
    "method": "get_blocks",
    "params": [[block_id]],  # bien [[block_id]] et PAS [block_id] ni [[, block_id]]
    "id": 0
}

headers = {"Content-Type": "application/json"}

resp = requests.post(api_url, data=json.dumps(payload), headers=headers)
print(resp.text)
